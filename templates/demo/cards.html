    <div class="mg-card-header">
        <h5 class="mg-card-title">Заголовок</h5>
        <p class="mg-card-subtitle">Подзаголовок</p>
    </div>
    <div class="mg-card-body">
        <p>Содержимое карточки</p>
    </div>
    <div class="mg-card-footer">
        <small class="text-muted">Футер</small>
    </div>
</div>

<!-- С цветом и размером -->
<div class="mg-card mg-card-primary mg-card-lg">
    <div class="mg-card-badge mg-card-badge-success">Посещен</div>
    <img src="image.jpg" class="mg-card-img" alt="Изображение">
    <div class="mg-card-body">
        <h5 class="mg-card-title">Заголовок</h5>
        <p>Содержимое</p>
    </div>
</div>

<!-- С круглым индикатором -->
<div class="mg-card">
    <div class="mg-card-status mg-card-status-success"></div>
    <div class="mg-card-body">
